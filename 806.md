var numberOfLines = function(widths, S) {
    let line = 1
    let next = 0
    for(let s of S){
        let index = s.charCodeAt() - 97
        let width = widths[index]
        if(next + width <= 100){
            next += width

        }else{
            line++
            next = width
        }
    }
    return [line,next]
};
