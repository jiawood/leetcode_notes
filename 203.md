* 主要思路是构建一个新的链表，移动新链表里面的指针。
```js
var removeElements = function(head, val) {
    let tep = new ListNode();
    tep.val = -1;
    tep.next = head;
    let tep1 = tep;

    while(tep.next){
        if(tep.next.val === val){
            tep.next = tep.next.next;
        }else{
            tep = tep.next;
        }

    }
    return tep1.next;
};
```
