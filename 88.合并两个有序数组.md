## 88.
* 因为不能使用额外的空间，所以在长数组num1上从后向前合并。从后向前比较，从后向前插入。
```js
var merge = function(nums1, m, nums2, n) {
    let current = m + n - 1;
    while(current>=0){
        if(n==0){
            return ;
        }
        if(m==0){
            nums1[current--] = nums2[--n];
            continue;
        }
        if(n==0){
            nums1[current--] = nums1[--m];
            continue;
        }
        if(nums1[m-1]>nums2[n-1]){
            nums1[current--] = nums1[--m];
        }else {
            nums1[current--] = nums2[--n]
        }
    }
    return nums1;
};
```
