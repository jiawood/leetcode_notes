* 一般的做法是双重循环，找出所有的子序列然后计算子序列的和，然后判断
* 有一种优化的办法是，用一个hash表来储存原数组中的累加值，其中两个累加值相减即为一个子序列之和
```js
var subarraySum = function(nums, k) {
    const hashmap = {}
    let acc = 0
    let count = 0
    for(let i = 0;i<nums.length;i++){
        acc += nums[i]
        if(acc === k){
            count++
        }
        if(hashmap[acc-k]!=void 0){
            count += hashmap[acc-k]
        }
        if(hashmap[acc]===void 0){
            hashmap[acc] = 1
        }else{
            hashmap[acc] += 1
        }
    }
    return count
};
```

第二次自己写的代码：

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
//使用一个hash表，记录前i项的和
var subarraySum = function(nums, k) {
    let hash = {}
    hash[0] = 1
    let sum = 0
    let count = 0
    for(let i in nums){
        sum += nums[i]
        if(hash[sum - k]){
            count = count + hash[sum - k]
        }
        if(!hash[sum]){
            hash[sum] = 1
        }else{
            hash[sum]++
        }
    }
    return  count
};
```

### 错误记录
* 本想用滑动窗口解题的，但最后发现是错的
* 滑动窗口 适合排序过的 数组来解题
* 本题数组是乱序的，滑动窗口不适合

错误代码：
```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var subarraySum = function(nums, k) {
    let n = nums.length
    let i = 0,j = 1
    let tep = [nums[0]]
    let res = []
    sum = nums[0]
    while(i < n){
        while(sum < k && j < n){
            tep.push(nums[j])
            sum += nums[j]
            j++
        }
        if(sum === k){
            res.push(tep.slice())
        }
        sum -= nums[i]
        tep.shift()
        i++
    }
    return res.length
};
```
