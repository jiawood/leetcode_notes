/**
 * @param {string} paragraph
 * @param {string[]} banned
 * @return {string}
 */
var mostCommonWord = function(paragraph, banned) {
    let str = paragraph.toLowerCase().replace(/[.?!,';]/g,' ').split(" ")
    let map = {}
    for(let s of str){
        if(banned.indexOf(s) == -1 && s != ' ' && s != ''){
            if(map[s]){
                map[s]++
            }else{
                map[s] = 1
            }
        }
    }
    let max = str[0]
    for(let i in map){
        if(map[i] > (!map[max] ? 0 : map[max]) && map[i] != " " && map[i] != ''){
            max = i
        }
    }
    return max
};
