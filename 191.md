* 组要是对位运算符的考察
#### 一般的办法
```js
var hammingWeight = function(n) {
    let count = 0;
    for(let i = 0;i < 32;i++){
        if(n&1==1){
            count+=1;
        }
        n = n>>>1;
    }
    return count;
```
#### tricks
* n & (n-1)  可以消除n里面的最后一个1.

```js
var hammingWeight = function(n) {
    let count = 0;
    while(n!=0){
        n=n&(n-1);
        count+=1;
    }
    return count;
};
```
