* 一般的做法是双重循环，找出所有的子序列然后计算子序列的和，然后判断
* 有一种优化的办法是，用一个hash表来储存原数组中的累加值，其中两个累加值相减即为一个子序列之和
```js
var subarraySum = function(nums, k) {
    const hashmap = {}
    let acc = 0
    let count = 0
    for(let i = 0;i<nums.length;i++){
        acc += nums[i]
        if(acc === k){
            count++
        }
        if(hashmap[acc-k]!=void 0){
            count += hashmap[acc-k]
        }
        if(hashmap[acc]===void 0){
            hashmap[acc] = 1
        }else{
            hashmap[acc] += 1
        }
    }
    return count
};
```
